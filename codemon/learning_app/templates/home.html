// CodeMon MVP Database Structure // Using DBML for ERD generation Table users { id integer [primary key] username varchar [unique, not null] email varchar [unique, not null] password varchar [not null] starter_pokemon_id integer rolls_available integer [default: 0] total_correct_answers integer [default: 0] created_at timestamp } Table pokemon { id integer [primary key] name varchar [not null] pokedex_number integer [unique] pokemon_type varchar [note: 'fire, water, grass, electric, etc.'] rarity varchar [note: 'common, uncommon, rare, legendary'] level integer [default: 1] image_url varchar description text created_at timestamp } Table questions { id integer [primary key] question_text text [not null] correct_answer varchar [not null] wrong_answer_1 varchar [not null] wrong_answer_2 varchar [not null] wrong_answer_3 varchar [not null] difficulty integer [note: '1-5 scale'] category varchar [note: 'programming, web_dev, databases, etc.'] created_at timestamp } Table quiz_attempts { id integer [primary key] user_id integer [not null] score integer [note: 'Number of correct answers'] total_questions integer earned_rolls integer [default: 0] completed_at timestamp } Table attempt_questions { id integer [primary key] quiz_attempt_id integer [not null] question_id integer [not null] user_answer varchar [note: 'What the user selected'] is_correct boolean answered_at timestamp } Table collections { id integer [primary key] user_id integer [not null] pokemon_id integer [not null] acquired_date timestamp is_favorite boolean [default: false] } // Relationships - No duplicates! Ref: users.starter_pokemon_id > pokemon.id Ref: collections.user_id > users.id Ref: collections.pokemon_id > pokemon.id Ref: quiz_attempts.user_id > users.id Ref: attempt_questions.quiz_attempt_id > quiz_attempts.id Ref: attempt_questions.question_id > questions.id // Many-to-many is already handled by the collections table relationships
